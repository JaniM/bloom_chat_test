# RULE 3
 
dead_peers <= (((heartbeat_timer * heartbeats) * peers).combos(heartbeats.addr => peers.addr) do |_, m, p|
  [m.addr] if m.isactive and (time >= (m.time + p.lifetime))
end)

die <= (stdio do |x|
  if (x.line[0, 5] == "/quit") then
    Thread.new do
      sleep(5.0)
      Kernel.exit(0)
    end
    [true]
  end
end)
